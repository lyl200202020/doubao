**机构表（sys_org）** - 如不存在则创建：

```sql
CREATE TABLE IF NOT EXISTS sys_org (
    org_id VARCHAR(32) PRIMARY KEY COMMENT '机构 ID（UUID）',
    org_code VARCHAR(32) NOT NULL UNIQUE COMMENT '机构代码（唯一）',
    org_name VARCHAR(128) NOT NULL COMMENT '机构名称',
    org_level TINYINT NOT NULL COMMENT '机构级别：0-总行，1-分行，2-支行，3-分理处',
    parent_org_id VARCHAR(32) COMMENT '上级机构 ID（顶级为 NULL）',
    org_status TINYINT NOT NULL DEFAULT 1 COMMENT '状态：0-停用，1-启用',
    sort_order INT NOT NULL DEFAULT 0 COMMENT '排序号',
    remark VARCHAR(500) COMMENT '备注',
    created_by VARCHAR(32) NOT NULL COMMENT '创建人',
    created_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_by VARCHAR(32) COMMENT '更新人',
    updated_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    deleted TINYINT NOT NULL DEFAULT 0 COMMENT '删除标记',
    INDEX idx_parent_org (parent_org_id),
    INDEX idx_org_level (org_level),
    INDEX idx_org_status (org_status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='机构表';
```

**用户表（sys_user）** - 如不存在则创建：


```java
package com.crm.domain.org.valueobject;

}
```

### 4.2 权限拦截器

```java
package com.crm.common.security.interceptor;

import com.crm.common.security.annotation.RequirePermission;
import com.crm.common.util.SecurityUtils;
import org.springframework.stereotype.Component;
import org.springframework.web.method.HandlerMethod;
import org.springframework.web.servlet.HandlerInterceptor;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.lang.reflect.Method;

/**
 * 权限拦截器
 */
@Component
public class PermissionInterceptor implements HandlerInterceptor {
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                           HttpServletResponse response, 
                           Object handler) throws Exception {
        // 不是方法调用，直接放行
        if (!(handler instanceof HandlerMethod)) {
            return true;
        }
        
        HandlerMethod handlerMethod = (HandlerMethod) handler;
        Method method = handlerMethod.getMethod();
        
        // 检查方法是否有权限注解
        RequirePermission requirePermission = method.getAnnotation(RequirePermission.class);
        if (requirePermission == null) {
            // 检查类上是否有权限注解
            requirePermission = handlerMethod.getBeanType()
                .getAnnotation(RequirePermission.class);
        }
        
        if (requirePermission != null) {
            // 获取当前用户权限
            var userPerms = SecurityUtils.getCurrentUserPermissions();
            
            // 校验权限
            if (!hasPermission(userPerms, requirePermission)) {
                response.setStatus(HttpServletResponse.SC_FORBIDDEN);
                response.setContentType("application/json;charset=UTF-8");
                response.getWriter().write("{\"code\":403,\"msg\":\"无权限\"}");
                return false;
            }
        }
        
        return true;
    }
    
    private boolean hasPermission(Set<String> userPerms, RequirePermission requirePermission) {
        if (userPerms == null || userPerms.isEmpty()) {
            return false;
        }
        
        String requiredPerm = requirePermission.value();
        
        // 超级管理员拥有所有权限
        if (userPerms.contains("*:*:*")) {
            return true;
        }
        
        if (RequirePermission.Logical.AND == requirePermission.logical()) {
            // AND 关系，需要所有权限
            String[] perms = requiredPerm.split(",");
            for (String perm : perms) {
                if (!userPerms.contains(perm.trim())) {
                    return false;
                }
            }
            return true;
        } else {
            // OR 关系，有一个权限即可
            String[] perms = requiredPerm.split(",");
            for (String perm : perms) {
                if (userPerms.contains(perm.trim())) {
                    return true;
                }
            }
            return false;
        }
    }
}
```

```java
@Service
public class OrgApplicationService {

    public OrgApplicationService(OrgRepository orgRepository) {
        this.orgRepository = orgRepository;

    /**
     * 获取机构树
     */
    @Transactional(readOnly = true)
    public List<OrgVO> getOrgTree() {

    /**
     * 新增机构
     */
    @Transactional
    public void addOrg(OrgDTO dto) {
        // 1. 校验机构代码唯一性
        if (!orgRepository.isOrgCodeUnique(null, dto.getOrgCode())) {
            throw new BusinessException("机构代码已存在");
        }
        
        // 2. 计算机构级别
        Integer parentLevel = 0;
        if (dto.getParentOrgId() != null) {
            var parentOrg = orgRepository.findById(dto.getParentOrgId())
                .orElseThrow(() -> new BusinessException("上级机构不存在"));
            parentLevel = parentOrg.getOrgLevel();
        }
        
        Integer newLevel = parentLevel + 1;
        if (newLevel > 3) {
            throw new BusinessException("机构级别最多 4 级");
        }
        
        // 3. 创建机构
        SysOrg org = new SysOrg();
        org.setOrgId(IdUtils.randomUUID());
        org.setOrgCode(dto.getOrgCode());
        org.setOrgName(dto.getOrgName());
        org.setOrgLevel(newLevel);
        org.setParentOrgId(dto.getParentOrgId());
        org.setOrgStatus(OrgStatus.ENABLED.getCode());
        org.setSortOrder(dto.getSortOrder());
        org.setRemark(dto.getRemark());
        
        orgRepository.save(org);
    }
    
    /**
     * 修改机构
     */
    @Transactional
    public void updateOrg(String orgId, OrgDTO dto) {
        // 1. 查询机构
        var org = orgRepository.findById(orgId)
            .orElseThrow(() -> new BusinessException("机构不存在"));
        
        // 2. 校验机构代码唯一性
        if (!orgRepository.isOrgCodeUnique(orgId, dto.getOrgCode())) {
            throw new BusinessException("机构代码已存在");
        }
        
        // 3. 更新机构
        org.setOrgCode(dto.getOrgCode());
        org.setOrgName(dto.getOrgName());
        org.setSortOrder(dto.getSortOrder());
        org.setRemark(dto.getRemark());
        
        orgRepository.save(org);
    }
    
    /**
     * 删除机构
     */
    @Transactional
    public void deleteOrg(String orgId) {
        // 1. 查询机构
        var org = orgRepository.findById(orgId)
            .orElseThrow(() -> new BusinessException("机构不存在"));
        
        // 2. 检查是否有下级机构
        if (orgRepository.hasChildren(orgId)) {
            throw new BusinessException("存在下级机构，无法删除");
        }
        
        // 3. 检查是否有关联用户
        if (orgRepository.hasUsers(orgId)) {
            throw new BusinessException("机构下有关联用户，无法删除");
        }
        
        // 4. 删除机构
        orgRepository.deleteById(orgId);
    }
    
    /**
     * 启用/停用机构
     */
    @Transactional
    public void toggleStatus(String orgId) {
        // 1. 查询机构
        var org = orgRepository.findById(orgId)
            .orElseThrow(() -> new BusinessException("机构不存在"));
        
        // 2. 切换状态
        Integer newStatus = org.getOrgStatus() == 1 ? 0 : 1;
        org.setOrgStatus(newStatus);
        
        // 3. 如果是停用，级联停用下级机构
        if (newStatus == 0) {
            cascadeDisableChildren(orgId);
        }
        
        orgRepository.save(org);
    }
    
    private List<OrgVO> buildOrgTree(List<SysOrg> allOrgs, String parentId) {
        return allOrgs.stream()
            .filter(org -> (parentId == null && org.getParentOrgId() == null) ||
                          (parentId != null && parentId.equals(org.getParentOrgId())))
            .map(org -> {
                OrgVO vo = convertToVO(org);
                List<OrgVO> children = buildOrgTree(allOrgs, org.getOrgId());
                vo.setChildren(children);
                return vo;
            })
            .collect(Collectors.toList());
    }
    
    private void cascadeDisableChildren(String orgId) {
        List<SysOrg> children = orgRepository.findByParentOrgId(orgId);
        for (SysOrg child : children) {
            child.setOrgStatus(0);
            orgRepository.save(child);
            cascadeDisableChildren(child.getOrgId());
        }
    }
    
    private OrgVO convertToVO(SysOrg org) {
        OrgVO vo = new OrgVO();
        vo.setOrgId(org.getOrgId());
        vo.setOrgCode(org.getOrgCode());
        vo.setOrgName(org.getOrgName());
        vo.setOrgLevel(org.getOrgLevel());
        vo.setParentOrgId(org.getParentOrgId());
        vo.setOrgStatus(org.getOrgStatus());
        vo.setSortOrder(org.getSortOrder());
        vo.setRemark(org.getRemark());
        return vo;
    }
```
