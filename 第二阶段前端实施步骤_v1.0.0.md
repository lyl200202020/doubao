## 二、权限控制实现（贯穿全程）

### 2.1 路由权限守卫

**文件路径**：`src/router/permission.ts`

**实施步骤**：

```typescript
import router from './index'
import { useUserStore } from '@/store/modules/user'
import { RouteRecordRaw } from 'vue-router'

const whiteList = ['/login', '/404'] // 白名单路由

router.beforeEach(async (to, from, next) => {
  const userStore = useUserStore()
  const hasToken = userStore.token
  
  if (hasToken) {
    if (to.path === '/login') {
      next({ path: '/' })
    } else {
      const hasRoles = userStore.roles && userStore.roles.length > 0
      
      if (hasRoles) {
        next()
      } else {
        try {
          // 获取用户信息和菜单
          const { menus, perms } = await userStore.getInfo()
          
          // 动态生成路由
          const accessRoutes = generateRoutes(menus)
          accessRoutes.forEach(route => {
            router.addRoute(route)
          })
          
          next({ ...to, replace: true })
        } catch (error) {
          // 获取失败，清空 token 并跳转登录页
          await userStore.resetToken()
          next(`/login?redirect=${to.path}`)
        }
      }
    }
  } else {
    // 没有 token
    if (whiteList.indexOf(to.path) !== -1) {
      next()
    } else {
      next(`/login?redirect=${to.path}`)
    }
  }
})

/**
 * 根据菜单生成路由
 */
function generateRoutes(menus: any[]): RouteRecordRaw[] {
  const routes: RouteRecordRaw[] = []
  
  menus.forEach(menu => {
    if (menu.menuType === 1) { // 目录
      const route: RouteRecordRaw = {
        path: menu.menuUrl,
        component: Layout,
        name: menu.menuId,
        meta: { title: menu.menuName, icon: menu.menuIcon },
        children: menu.children ? generateRoutes(menu.children) : []
      }
      routes.push(route)
    } else if (menu.menuType === 2) { // 菜单
      const route: RouteRecordRaw = {
        path: menu.menuUrl,
        component: () => import(`@/views${menu.menuUrl}/index.vue`),
        name: menu.menuId,
        meta: { title: menu.menuName, icon: menu.menuIcon, perms: menu.perms }
      }
      routes.push(route)
    }
  })
  
  return routes
}
```

### 2.2 动态菜单生成

**文件路径**：`src/store/modules/app.ts`

**实施步骤**：

```typescript
import { defineStore } from 'pinia'
import { getMenus } from '@/api/sys/menu'

export interface Menu {
  menuId: string
  menuName: string
  menuType: number
  menuUrl: string
  menuIcon: string
  perms?: string
  children?: Menu[]
}

export const useAppStore = defineStore('app', {
  state: () => ({
    menus: [] as Menu[],
    permissions: [] as string[]
  }),
  
  getters: {
    // 获取侧边栏菜单
    sidebarMenus(): Menu[] {
      return this.menus.filter(menu => menu.menuType === 1)
    },
    
    // 获取所有权限标识
    allPermissions(): string[] {
      return this.permissions
    }
  },
  
  actions: {
    async generateMenus() {
      const res = await getMenus()
      this.menus = res.data
      
      // 提取所有权限标识
      this.permissions = this.extractPermissions(this.menus)
      
      return this.menus
    },
    
    extractPermissions(menus: Menu[]): string[] {
      const perms: string[] = []
      menus.forEach(menu => {
        if (menu.perms) {
          perms.push(menu.perms)
        }
        if (menu.children && menu.children.length > 0) {
          perms.push(...this.extractPermissions(menu.children))
        }
      })
      return perms
    }
  }
})
```

### 2.3 按钮权限指令

**文件路径**：`src/directives/permission.ts`

**实施步骤**：

```typescript
import { useUserStore } from '@/store/modules/user'
import { Directive, DirectiveBinding } from 'vue'

export const permission: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding) {
    const { value } = binding
    const userStore = useUserStore()
    const permissions = userStore.permissions
    
    if (value && value instanceof Array && value.length > 0) {
      const permissionFlag = value
      
      const hasPerm = permissions.some(perm => {
        return permissionFlag.includes(perm)
      })
      
      if (!hasPerm) {
        el.parentNode && el.parentNode.removeChild(el)
      }
    } else {
      throw new Error('请设置权限标签值')
    }
  }
}
```

**全局注册**：`src/main.ts`

              v-permission="['sys:org:edit']"
              type="warning"
              link
              @click="handleToggleStatus(row)"
            >
              {{ row.orgStatus === 1 ? '停用' : '启用' }}
            </el-button>
            <el-button
              v-permission="['sys:org:delete']"
              type="danger"
              link
              @click="handleDelete(row)"
            >
              删除
            </el-button>
          </template>
        </el-table-column>
      </el-table>
    </el-card>
    
    <!-- 机构表单弹窗 -->

import { Org, OrgDTO } from '@/types/org'
import OrgForm from './components/OrgForm.vue'

const formVisible = ref(false)

  loadOrgTree()

  const res = await getOrgTree()
  orgTree.value = res.data

  formData.value = null
  formVisible.value = true

  formData.value = { ...row }
  formVisible.value = true

    await updateOrg(formData.value.orgId, data)

    loadOrgTree()

  await toggleStatus(row.orgId)
  ElMessage.success('操作成功')
  loadOrgTree()

  return levels[level] || ''

  const types = ['danger', 'warning', 'success', 'info']
  return types[level] || ''

### 3.4 机构表单组件

**文件路径**：`src/views/sys/org/components/OrgForm.vue`

**实施步骤**：

```vue
<template>
  <el-dialog
    :model-value="modelValue"
    :title="formData?.orgId ? '编辑机构' : '新增机构'"
    width="500px"
    @close="handleClose"
  >
    <el-form
      ref="formRef"
      :model="form"
      :rules="rules"
      label-width="100px"
    >
      <el-form-item label="上级机构" prop="parentOrgId">
        <el-tree-select
          v-model="form.parentOrgId"
          :data="orgTree"
          :props="{ label: 'orgName', value: 'orgId', children: 'children' }"
          check-strictly
          placeholder="请选择上级机构（顶级机构不选）"
          style="width: 100%"
        />
      </el-form-item>
      
      <el-form-item label="机构代码" prop="orgCode">
        <el-input v-model="form.orgCode" placeholder="请输入机构代码" />
      </el-form-item>
      
      <el-form-item label="机构名称" prop="orgName">
        <el-input v-model="form.orgName" placeholder="请输入机构名称" />
      </el-form-item>
      
      <el-form-item label="排序号" prop="sortOrder">
        <el-input-number v-model="form.sortOrder" :min="0" :max="999" />
      </el-form-item>
      
      <el-form-item label="备注" prop="remark">
        <el-input
          v-model="form.remark"
          type="textarea"
          :rows="3"
          placeholder="请输入备注"
        />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="handleClose">取消</el-button>
      <el-button type="primary" @click="handleSubmit" :loading="loading">
        确定
      </el-button>
    </template>
  </el-dialog>
</template>

<script setup lang="ts">
import { ref, reactive, watch } from 'vue'
import { ElForm, FormRules } from 'element-plus'
import { Org, OrgDTO } from '@/types/org'

interface Props {
  modelValue: boolean
  formData: OrgDTO | null
  orgTree: Org[]
}

const props = withDefaults(defineProps<Props>(), {
  modelValue: false,
  formData: null,
  orgTree: () => []
})

const emit = defineEmits<{
  (e: 'update:modelValue', value: boolean): void
  (e: 'submit', data: OrgDTO): void
}>()

const formRef = ref<InstanceType<typeof ElForm>>()
const loading = ref(false)

const form = reactive<OrgDTO>({
  orgCode: '',
  orgName: '',
  parentOrgId: undefined,
  sortOrder: 0,
  remark: ''
})

const rules: FormRules = {
  orgCode: [{ required: true, message: '请输入机构代码', trigger: 'blur' }],
  orgName: [{ required: true, message: '请输入机构名称', trigger: 'blur' }]
}

watch(() => props.formData, (val) => {
  if (val) {
    Object.assign(form, val)
  }
}, { immediate: true, deep: true })

function handleClose() {
  emit('update:modelValue', false)
  formRef.value?.resetFields()
}

function handleSubmit() {
  formRef.value?.validate(async (valid) => {
    if (valid) {
      loading.value = true
      try {
        emit('submit', { ...form })
      } finally {
        loading.value = false
      }
    }
  })
}
</script>
```
